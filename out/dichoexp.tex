\begin{algorithm}[H]
\SetKwInOut{Input}{input}
\SetKwInOut{Output}{output}
\Input{a:int,
b:int}
\Output{int}
\If{ $ b = 0 $ }{
  \Return $ 1 $\;
}

\If{ $ (b mod 2) = 0 $ }{
  $int\:o := exp(a, b \div 2)$\;
  \Return $ o * o $\;
}
\Else{
  \Return $ a * exp(a, b - 1) $\;
}
\caption{exp}
\end{algorithm}
