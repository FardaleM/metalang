\begin{algorithm}[H]
\SetKwInOut{Input}{input}
\SetKwInOut{Output}{output}
\Input{str:array<char>,
len:int}
\Output{int}
$array<int>\:stack=\:new\:int[len]$\;
\ForEach{ $i\;indice\;of\;stack$ }{$stack[i]:=0$}
$int\:ptrStack := 0$\;
$int\:ptrStr := 0$\;
\While{$ ptrStr < len $}{
  \If{ $ str[ptrStr] = " " $ }{
    ptrStr ++;
  }
  \Else{
    \If{ $ is_number(str[ptrStr]) $ }{
      $int\:num := 0$\;
      \While{$ str[ptrStr] \neq " " $}{
        $num \leftarrow num * 10 + str[ptrStr] - "0" $\;
        ptrStr ++;
      }
      $stack[ptrStack] \leftarrow num $\;
      ptrStack ++;
    }
    \Else{
      \If{ $ str[ptrStr] = "+" $ }{
        $stack[ptrStack - 2] \leftarrow stack[ptrStack - 2] + stack[ptrStack - 1] $\;
        ptrStack --;
        ptrStr ++;
      }
      
    }
  }
}
\Return $ stack[0] $\;
\caption{npi}
\end{algorithm}
