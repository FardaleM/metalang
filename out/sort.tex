\begin{algorithm}[H]

\SetKwInOut{Output}{output}
\Input{tab:array<int>,
len:int}
\Output{void}
\For{ $i\;:=\;0\; to\;len - 1$}{
                                                         \For{ $j\;:=\;i + 1\; to\;len - 1$}{
                                                         
                                                           \If{ $ tab[i] > tab[j] $ }{
                                                             $int\:tmp := tab[i]$\;
                                                             $tab[i] \leftarrow tab[j] $\;
                                                             $tab[j] \leftarrow tmp $\;
                                                           }
                                                           }
                                                         }

\caption{sort\_}
\end{algorithm}
