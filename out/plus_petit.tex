\begin{algorithm}[H]
\SetKwInOut{Input}{input}
\SetKwInOut{Output}{output}
\Input{tab:array<int>,
a:int,
b:int}
\Output{int}
$int\:m := (a + b) \div 2$\;
\If{ $ a = m $ }{
  \If{ $ tab[a] = m $ }{
    \Return $ b $\;
  }
  \Else{
    \Return $ a $\;
  }
}

$int\:i := a$\;
$int\:j := b$\;
\While{$ i < j $}{
  $int\:e := tab[i]$\;
  \If{ $ e < m $ }{
    i ++;
  }
  \Else{
    j --;
    $tab[i] \leftarrow tab[j] $\;
    $tab[j] \leftarrow e $\;
  }
}
\If{ $ i < m $ }{
  \Return $ go(tab, a, m) $\;
}
\Else{
  \Return $ go(tab, m, b) $\;
}
\caption{go}
\end{algorithm}
