\begin{algorithm}[H]
\SetKwInOut{Input}{input}
\SetKwInOut{Output}{output}
\Input{lst:array<int>,
len:int}
\Output{int}
$int\:current := 0$\;
$int\:max := 0$\;
\For{ $i\;:=\;0\; to\;len - 1$}{
                                  current += $ lst[i] $;
                                  \If{ $ current < 0 $ }{
                                    $current \leftarrow 0 $\;
                                  }
                                  
                                  \If{ $ max < current $ }{
                                    $max \leftarrow current $\;
                                  }
                                  }

\Return $ max $\;
\caption{summax}
\end{algorithm}
