\begin{algorithm}[H]
\SetKwInOut{Input}{input}
\SetKwInOut{Output}{output}
\Input{n:int,
passepartout:array<array<int>>, m:int,
serrures:array<array<int>>}
\Output{int}
$int\:max\_ancient := 0$\;
$int\:max\_recent := 0$\;
\For{ $i\;:=\;0\; to\;m - 1$}{
                                \If{ $ serrures[i][0] = -1 and serrures[i][1] > max\_ancient $ }{
                                  $max\_ancient \leftarrow serrures[i][1] $\;
                                }
                                
                                \If{ $ serrures[i][0] = 1 and serrures[i][1] > max\_recent $ }{
                                  $max\_recent \leftarrow serrures[i][1] $\;
                                }
                                }

$int\:max\_ancient\_pp := 0$\;
$int\:max\_recent\_pp := 0$\;
\For{ $i\;:=\;0\; to\;n - 1$}{
                                $array<int>\:pp := passepartout[i]$\;
                                \If{ $ pp[0] \geq max\_ancient and pp[1] \geq max\_recent $ }{
                                  \Return $ 1 $\;
                                }
                                
                                $max\_ancient\_pp \leftarrow max2(max\_ancient\_pp, pp[0]) $\;
                                $max\_recent\_pp \leftarrow max2(max\_recent\_pp, pp[1]) $\;}

\If{ $ max\_ancient\_pp \geq max\_ancient and max\_recent\_pp \geq max\_recent $ }{
  \Return $ 2 $\;
}
\Else{
  \Return $ 0 $\;
}
\caption{nbPassePartout}
\end{algorithm}
