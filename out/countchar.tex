\begin{algorithm}[H]
\SetKwInOut{Input}{input}
\SetKwInOut{Output}{output}
\Input{tab:array<char>,
tofind:char,
len:int}
\Output{int}
$int\:out := 0$\;
\For{ $i\;:=\;0\; to\;len - 1$}{
                                  \If{ $ tab[i] = tofind $ }{
                                    out ++;
                                  }
                                  }

\Return $ out $\;
\caption{nth}
\end{algorithm}
