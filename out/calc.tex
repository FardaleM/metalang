\begin{algorithm}[H]
\SetKwInOut{Input}{input}
\SetKwInOut{Output}{output}
\Input{a:int,
b:int,
i:int}
\Output{int}
$int\:out\_ := 0$\;
$int\:a2 := a$\;
$int\:b2 := b$\;
\For{ $j\;:=\;0\; to\;i + 1$}{
                                $print_{int}$($ j $)\;
                                out\_ += $ a2 $;
                                $int\:tmp := b2$\;
                                b2 += $ a2 $;
                                $a2 \leftarrow tmp $\;}

\Return $ out\_ $\;
\caption{fibo}
\end{algorithm}
