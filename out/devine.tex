\begin{algorithm}[H]
\SetKwInOut{Input}{input}
\SetKwInOut{Output}{output}
\Input{nombre:int,
tab:array<int>,
len:int}
\Output{bool}
$int\:min := tab[0]$\;
$int\:max := tab[1]$\;
\For{ $i\;:=\;2\; to\;len - 1$}{
                                  \If{ $ tab[i] > max or tab[i] < min $ }{
                                    \Return $ false $\;
                                  }
                                  
                                  \If{ $ tab[i] < nombre $ }{
                                    $min \leftarrow tab[i] $\;
                                  }
                                  
                                  \If{ $ tab[i] > nombre $ }{
                                    $max \leftarrow tab[i] $\;
                                  }
                                  
                                  \If{ $ tab[i] = nombre and len \neq i + 1 $ }{
                                    \Return $ false $\;
                                  }
                                  }

\Return $ true $\;
\caption{devine}
\end{algorithm}
