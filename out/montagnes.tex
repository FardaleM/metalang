\begin{algorithm}[H]
\SetKwInOut{Input}{input}
\SetKwInOut{Output}{output}
\Input{tab:array<int>,
len:int}
\Output{int}
$int\:max := 1$\;
$int\:j := 1$\;
$int\:i := len - 2$\;
\While{$ i \geq 0 $}{
  $int\:x := tab[i]$\;
  \While{$ j \geq 0 and x > tab[len - j] $}{
    j --;
  }
  j ++;
  $tab[len - j] \leftarrow x $\;
  \If{ $ j > max $ }{
    $max \leftarrow j $\;
  }
  
  i --;
}
\Return $ max $\;
\caption{montagnes}
\end{algorithm}
